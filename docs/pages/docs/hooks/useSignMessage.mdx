# useSignMessage

Hook for signing messages with connected account.

```tsx
import { useSignMessage } from 'wagmi'
```

## Usage

The following examples use the message: `gm wagmi frens`.

```tsx
import { useSignMessage } from 'wagmi'

const App = () => {
  const [{ data, error, loading }, signMessage] = useSignMessage({
    message: 'gm wagmi frens',
  })

  if (!data)
    return (
      <button disabled={loading} onClick={async () => await signMessage()}>
        Sign message
      </button>
    )

  return (
    <div>
      {data && <div>Signature: {data}</div>}
      {error && <div>Error signing message</div>}
    </div>
  )
}
```

## Return Values

### result

```ts
{
  data?: string
  error?: Error
  loading?: boolean
}
```

### signMessage

```ts
(config?: { message: string }) => Promise<string | Error>
```

## Configuration

### message (optional)

Message to sign with wallet.

```tsx highlight='5'
import { useSignMessage } from 'wagmi'

const App = () => {
  const [{ data, error, loading }, signMessage] = useSignMessage({
    message: 'gm wagmi frens',
  })

  return ...
}
```
